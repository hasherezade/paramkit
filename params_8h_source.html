234  bool parse(int argc, T_CHAR* argv[])
235  {
236  bool paramHelp = false;
237  size_t count = 0;
238  for (int i = 1; i < argc; i++) {
239  std::string param_str = to_string(argv[i]);
240  if (!isParam(param_str)) {
241  printUnknownArgument(param_str);
242  continue;
243  }
244  bool found = false;
245  param_str = skipParamPrefix(param_str);
246 
247  std::map<std::string, Param*>::iterator itr;
248  for (itr = myParams.begin(); itr != myParams.end(); ++itr) {
249  Param *param = itr->second;
250  if (param_str == PARAM_HELP2 || param_str == PARAM_HELP1) {
251  if (param_str == PARAM_HELP2) {
252  const bool hasArg = (i + 1) < argc && !(isParam(to_string(argv[i + 1])));
253  if (hasArg) {
254  const std::string nextVal = to_string(argv[i + 1]);
255  printHelp(nextVal, true);
256  return false;
257  }
258  }
259  const bool shouldExpand = (param_str == PARAM_HELP1) ? false : true;
260  printHelp("", shouldExpand);
261  return false;
262  }
263  if (this->versionStr.length()) {
264  if (param_str == PARAM_VERSION || param_str == PARAM_VERSION2) {
265  this->printVersionInfo();
266  return false;
267  }
268  }
269  if (param_str == param->argStr) {
270  if (!param->isActive()) {
271  paramkit::print_in_color(RED, "WARNING: chosen inactive parameter: " + param_str + "\n");
272  }
273  // has an argument:
274  const bool hasArg = (i + 1) < argc &&
275  ( param->requiredArg || !(isParam(to_string(argv[i + 1]))) );
276  if (hasArg) {
277  const std::string nextVal = to_string(argv[i + 1]);
278  i++; // icrement index: move to the next argument
279  found = true;
280  bool isParsed = false;
281 
282  if (nextVal == PARAM_HELP1) {
283  paramHelp = true;
284  isParsed = true;
285  }
286  else {
287  isParsed = param->parse(nextVal.c_str());
288  if (!isParsed) paramHelp = true;
289  }
290 
291  //help requested explicitly or parsing failed
292  if (paramHelp) {
293  if (!isParsed) {
294  paramkit::print_in_color(RED, "Parsing the parameter failed. Correct options:\n");
295  }
296  paramkit::print_in_color(RED, param_str);
297  param->printDesc();
298  break;
299  }
300  break;
301  }
302  // does not require an argument:
303  if (!param->requiredArg) {
304  param->parse((char*)nullptr);
305  found = true;
306  break;
307  }
308  // requires an argument, but it is missing:
309  paramkit::print_in_color(RED, param_str);
310  paramHelp = true;
311  param->printDesc();
312  found = true;
313  break;
314  }
315  }
316  if (found) {
317  count++;
318  }
319  else {
320  printUnknownParam(param_str);
321  print_in_color(HILIGHTED_COLOR, "Similar parameters:\n");
322  this->printInfo(false, param_str, true);
323  return false;
324  }
325  }
326  if (paramHelp) {
327  return false;
328  }
329  if (!this->hasRequiredFilled()) {
330  print_in_color(WARNING_COLOR, "Missing required parameters:\n");
331  this->printInfo(true, "", true);
332  return false;
333  }
334  if (this->countCategory(true) == 0 && countFilled(false) == 0) {
335  std::stringstream ss1;
336  ss1 << "Run with parameter " << PARAM_SWITCH1 << PARAM_HELP1 << " or " << PARAM_SWITCH1 << PARAM_HELP2 << " to see the options...\n";
337  print_in_color(YELLOW, ss1.str());
338  }
339  return true;
